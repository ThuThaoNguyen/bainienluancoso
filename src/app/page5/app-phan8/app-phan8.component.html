Ví dụ: hãy cho biết mã nhân viên của các nhân viên là thư ký hoặc có trình độ B Anh văn .

  SELECT DISTINCT a . many

  FROM nhanvien AS a , tdonn AS b

  WHERE a . Congviec = " Thu ky ' or (a . many = b . many AD b . ngoaingu = ' Anh van ' AND b . bangcap = ' B ' )
